def rozmiar_kompresji(T):
    n = len(T)
    b = 0
    i = 0

    while i < n:
        licznik = 1

        while i + licznik < n and T[i] == T[i + licznik]:
            licznik += 1

        b += 2
        i += licznik
    return b
