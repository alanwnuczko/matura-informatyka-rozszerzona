```python
def suma_cyfr(n):
    s = 0

    while n > 0:
        cyfra = n % 10
        s = s + cyfra
        n //= 10

    return s

def oblicz_k(n):
    k = n - suma_cyfr(n)
    k //= 9
    return k

print(oblicz_k(375))
```
